#include int main() { int numbers[] = {5, 2, 8, 1, 9, 3}; int size = 6; int i, j, temp; printf("Original array: "); for (i = 0; i < size; i++) { printf("%d ", numbers[i]); } printf("\n"); for (i = 0; i < size - 1; i++) { for (j = 0; j < size - 1 - i; j++) { if (numbers[j] > numbers[j + 1]) { // Swap numbers[j] and numbers[j + 1] temp = numbers[j]; numbers[j] = numbers[j + 1]; numbers[j + 1] = temp; } } } // Print sorted array printf("Sorted array in ascending order: "); for (i = 0; i < size; i++) { printf("%d ", numbers[i]); } printf("\n"); return 0; }
